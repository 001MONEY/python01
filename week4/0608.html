#  任务3: 语文与英语成绩相关性分析（散点图）

# 提取语、英成绩和学生姓名

fig, ax = plt.subplots(figsize=(10, 6))

# scatter(x, y) —— 散点图，s 控制点大小，c 控制颜色，alpha 透明度
# zorder 控制图层顺序，越大越靠上

# xytext=(5,5) 表示文本偏移 (5px, 5px)

# polyfit 做一次线性拟合（1 次多项式），poly1d 构建多项式函数

p = np.poly1d(z)

x_fit = np.linspace(min(chinese)-2, max(chinese)+2, 50)
ax.plot(x_fit, p(x_fit), "r--", lw=2, alpha=0.7,
        label=f"拟合线 (r={corr:.3f})")  # r 值显示在图例中

ax.set_title("语文与英语成绩相关性分析", fontsize=15)
ax.set_xlabel("语文成绩")

# ==========================================================
#  数据：矩形用 [x1, y1, x2, y2] 表示（左下角, 右上角）
#  —— 采用标准笛卡尔坐标系，y 轴向上为正

box   = np.array([ 2,  2, 20, 15])   # 候选框 (candidate)
boxes = np.array([15, 12, 25, 21])   # 真实框 (ground truth)

# ==========================================================
#  1. 画出两个矩形
# ==========================================================
fig, ax = plt.subplots(figsize=(6, 6))

# box 矩形 (蓝色)

                       fill=False, edgecolor='blue', lw=3, label='Candidate')
ax.add_patch(rect1)

# boxes 矩形 (红色)
rect2 = plt.Rectangle((boxes[0], boxes[1]), boxes[2]-boxes[0], boxes[3]-boxes[1],
                       fill=False, edgecolor='red', lw=3, label='Ground Truth')
ax.add_patch(rect2)

# 标注四个角坐标
ax.text(box[0], box[1], f'({box[0]},{box[1]})', fontsize=9, ha='right', va='top', color='blue')
ax.text(box[2], box[3], f'({box[2]},{box[3]})', fontsize=9, ha='left', va='bottom', color='blue')
ax.text(boxes[0], boxes[1], f'({boxes[0]},{boxes[1]})', fontsize=9, ha='right', va='top', color='red')
ax.text(boxes[2], boxes[3], f'({boxes[2]},{boxes[3]})', fontsize=9, ha='left', va='bottom', color='red')

ax.set_xlim(0, 30); ax.set_ylim(0, 25)
ax.set_aspect('equal')   # 保持纵横比一致

ax.legend(); ax.grid(alpha=0.3)
ax.set_title('Two Rectangles (Cartesian Coordinates)')
plt.show()

# ==========================================================
#  5. 求出前3个最相似特征的姓名
#  数据：file/target_feat.txt（目标特征）
#        file/feat.txt（特征库，格式：id|姓名|[特征向量]）
#  方法：余弦相似度 (cosine similarity)
# ==========================================================
import numpy as np

# ---------- 读取目标特征 ----------
with open('file/target_feat.txt', 'r', encoding='utf-8') as f:
    target_str = f.read().strip()
target_feat = np.array(eval(target_str))  # 字符串转数组

# ---------- 读取特征库 ----------
names = []       # 存储姓名
features = []    # 存储特征向量
with open('file/feat.txt', 'r', encoding='utf-8') as f:
    for line in f:
        line = line.strip()
        if not line:
            continue
        # 格式：id|姓名|[特征向量]
        parts = line.split('|')
        idx = parts[0]
        name = parts[1]
        feat_str = '|'.join(parts[2:])   # 特征向量部分
        feat = np.array(eval(feat_str))
        names.append(name)
        features.append(feat)

features = np.array(features)

print(f"目标特征维度: {target_feat.shape}")
print(f"特征库总数: {len(names)}")

# ---------- 计算余弦相似度 ----------
# 余弦相似度 = (A·B) / (||A|| * ||B||)
target_norm = np.linalg.norm(target_feat)
features_norm = np.linalg.norm(features, axis=1)
dot_products = np.dot(features, target_feat)
similarities = dot_products / (features_norm * target_norm + 1e-8)

# ---------- 找出目标自身（相似度 ≈ 1.0） ----------
self_idx = np.argmax(similarities)
print(f"\n目标自身: id={self_idx}, name='{names[self_idx]}', similarity={similarities[self_idx]:.4f}")

# ---------- 排除自身，取前3 ----------
top_k = 3
similarities_copy = similarities.copy()
similarities_copy[self_idx] = -1  # 自身置为最低
top_indices = np.argsort(similarities_copy)[::-1][:top_k]

print(f"\n🔝 前{top_k}个最相似特征（排除自身）:")
print(f"{'排名':<6}{'姓名':<10}{'相似度':<10}{'原索引'}")
print("-" * 36)
for rank, idx in enumerate(top_indices, 1):
    print(f"{rank:<6}{names[idx]:<10}{similarities[idx]:.4f}    (id={idx})")
